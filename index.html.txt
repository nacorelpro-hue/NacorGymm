    function switchTab(tab) {
      currentActiveTab = tab;
      ['entrenar', 'ejercicios', 'stats'].forEach(t => {
        const btn = document.getElementById(`nav-${t}`);
        if (t === tab) {
          btn.classList.add('text-brand');
          btn.classList.remove('text-zinc-500');
        } else {
          btn.classList.remove('text-brand');
          btn.classList.add('text-zinc-500');
        }
      });

    }

    function renderTabContent() {
      const container = document.getElementById('app-content');
      if (currentActiveTab === 'entrenar') {
        if (activeWorkout) {
          renderActiveWorkout(container);
        } else {
          renderWorkoutSelection(container);
        }
      } else if (currentActiveTab === 'ejercicios') {
        renderExerciseList(container);
      } else if (currentActiveTab === 'stats') {
        renderStats(container);
      }

    // --- ENTRENAMIENTO ACTIVO ---
    async function startWorkout(routineName) {
      const exercises = DEFAULT_ROUTINES[routineName];
      activeWorkout = {
        routine: routineName,
        date: new Date().toISOString(),
        exercises: []
      };

      for (let ex of exercises) {
        const lastSession = await getLastExerciseSession(ex);
        activeWorkout.exercises.push({
          name: ex,
          sets: lastSession ? lastSession.sets.map(s => ({ peso: s.peso, reps: s.reps, done: false })) : [{ peso: '', reps: '', done: false }]
        });
      }
      renderTabContent();
    }

    function addSet(exIdx) {
      const sets = activeWorkout.exercises[exIdx].sets;
      const lastSet = sets[sets.length - 1] || { peso: '', reps: '' };
      sets.push({ peso: lastSet.peso, reps: lastSet.reps, done: false });
      renderTabContent();
    }

    function deleteSet(exIdx, setIdx) {
      if (activeWorkout.exercises[exIdx].sets.length > 1) {
        activeWorkout.exercises[exIdx].sets.splice(setIdx, 1);
        renderTabContent();
      }
    }

    function toggleSetDone(exIdx, setIdx) {
      const set = activeWorkout.exercises[exIdx].sets[setIdx];
      const pesoVal = document.getElementById(`peso-${exIdx}-${setIdx}`).value;
      const repsVal = document.getElementById(`reps-${exIdx}-${setIdx}`).value;

      set.peso = pesoVal !== '' ? parseFloat(pesoVal) : '';
      set.reps = repsVal !== '' ? parseInt(repsVal) : '';
      set.done = !set.done;
      
      if (set.done) {
        startTimer(90); // Inicia temporizador de 90 segundos al completar serie
      }
      renderTabContent();
    }

    // --- TEMPORIZADOR DE DESCANSO INTELIGENTE ---
    function startTimer(seconds) {
      clearInterval(timerInterval);
      const banner = document.getElementById('timer-banner');
      const display = document.getElementById('timer-display');
      banner.classList.remove('hidden');

      let timeRemaining = seconds;
      timerInterval = setInterval(() => {
        const mins = Math.floor(timeRemaining / 60).toString().padStart(2, '0');
        const secs = (timeRemaining % 60).toString().padStart(2, '0');
        display.textContent = `${mins}:${secs}`;
        
        if (timeRemaining <= 0) {
          clearInterval(timerInterval);
          banner.classList.add('hidden');
          if (navigator.vibrate) navigator.vibrate([200, 100, 200]); // Vibración sutil
        }
        timeRemaining--;
      }, 1000);
    }

    function stopTimer() {
      clearInterval(timerInterval);
      document.getElementById('timer-banner').classList.add('hidden');
    }

    async function finishWorkout() {
      const workoutToSave = {
        routine: activeWorkout.routine,
        date: new Date().toLocaleDateString('es-ES', { day: 'numeric', month: 'short', year: 'numeric' }),
        rawDate: new Date(),
        exercises: activeWorkout.exercises.map(ex => ({
          name: ex.name,
          sets: ex.sets.filter(s => s.done)
        })).filter(ex => ex.sets.length > 0)
      };

      if (workoutToSave.exercises.length === 0) {
        alert("Debes completar al menos una serie (✓).");
        return;
      }

      const transaction = db.transaction(["workouts", "exercises_history"], "readwrite");
      transaction.objectStore("workouts").add(workoutToSave);

      workoutToSave.exercises.forEach(ex => {
        transaction.objectStore("exercises_history").add({
          exerciseName: ex.name,
          date: workoutToSave.date,
          rawDate: workoutToSave.rawDate,
          sets: ex.sets
        });
      });

      activeWorkout = null;
      stopTimer();
      alert("¡Entrenamiento guardado!");
      switchTab('entrenar');
    }

    function cancelWorkout() {
      if (confirm("¿Descartar este entrenamiento?")) {
        activeWorkout = null;
        stopTimer();
        switchTab('entrenar');
      }
    }

    function getLastExerciseSession(exerciseName) {
      return new Promise((resolve) => {
        const transaction = db.transaction(["exercises_history"], "readonly");
        const store = transaction.objectStore("exercises_history");
        let result = null;

        store.openCursor(null, "prev").onsuccess = (e) => {
          const cursor = e.target.result;
          if (cursor) {
            if (cursor.value.exerciseName === exerciseName) {
              result = cursor.value;
              resolve(result);
              return;
            }
            cursor.continue();
          } else {
            resolve(null);
          }
        };
      });
    }

    // --- DETALLE E HISTORIAL DEL EJERCICIO ---
    async function renderExerciseHistory(exerciseName) {
      const container = document.getElementById('app-content');
      const rawHistory = await getExerciseHistory(exerciseName);
      const history = rawHistory.reverse();

      let recordMax = 0;
      let totalVolume = 0;
      let maxReps = 0;
      let listHTML = '';
      const chartLabels = [];
      const chartData = [];

      history.forEach((session) => {
        let maxWeight = 0;
        let sessionHTML = `
          <div class="bg-zinc-900 border border-zinc-800 p-4 rounded-xl mb-2">
            <div class="flex justify-between items-center mb-2 pb-1.5 border-b border-zinc-800/50">
              <span class="text-[10px] font-bold text-zinc-400">${session.date}</span>
            </div>
        `;

        session.sets.forEach((set, sIdx) => {
          const weight = parseFloat(set.peso) || 0;
          const reps = parseInt(set.reps) || 0;
          if (weight > recordMax) recordMax = weight;
          if (weight > maxWeight) maxWeight = weight;
          if (reps > maxReps) maxReps = reps;
          totalVolume += weight * reps;

          sessionHTML += `
            <div class="flex justify-between text-xs py-1">
              <span class="text-zinc-500">Serie ${sIdx + 1}</span>
              <span class="font-semibold text-zinc-200">${set.peso} kg × ${set.reps} reps</span>
            </div>
          `;
        });

        sessionHTML += `</div>`;
        listHTML += sessionHTML;

        chartLabels.push(session.date);
        chartData.push(maxWeight);
      });

      // Cálculo del 1RM Estimado usando la fórmula de Epley
      const calculated1RM = recordMax > 0 && maxReps > 0 ? Math.round(recordMax * (1 + maxReps / 30)) : 0;

      container.innerHTML = `
        <button onclick="switchTab('ejercicios')" class="text-brand text-xs font-extrabold mb-4 flex items-center gap-1">← Volver</button>
        <h2 class="text-xl font-extrabold mb-3">${exerciseName}</h2>
        
        <div class="grid grid-cols-3 gap-2 mb-4">
          <div class="bg-zinc-900 border border-zinc-800 p-3 rounded-xl text-center">
            <span class="text-[8px] text-zinc-400 font-bold block mb-0.5">Récord</span>
            <span class="text-base font-extrabold text-brand">${recordMax} Kg</span>
          </div>
          <div class="bg-zinc-900 border border-zinc-800 p-3 rounded-xl text-center">
            <span class="text-[8px] text-zinc-400 font-bold block mb-0.5">1RM Est.</span>
            <span class="text-base font-extrabold text-white">${calculated1RM} Kg</span>
          </div>
          <div class="bg-zinc-900 border border-zinc-800 p-3 rounded-xl text-center">
            <span class="text-[8px] text-zinc-400 font-bold block mb-0.5">Vol. Total</span>
            <span class="text-base font-extrabold text-zinc-300">${totalVolume} Kg</span>
          </div>
        </div>

        <div class="bg-zinc-900 border border-zinc-800 p-4 rounded-xl mb-4">
          <canvas id="exerciseChart" class="w-full"></canvas>
        </div>

        <h3 class="text-xs font-bold text-zinc-400 uppercase tracking-wider mb-2">Historial</h3>
        ${listHTML || '<p class="text-xs text-zinc-500">Sin entrenamientos previos.</p>'}
      `;

      if (chartData.length > 0) {
        const ctx = document.getElementById('exerciseChart').getContext('2d');
        new Chart(ctx, {
          type: 'line',
          data: {
            labels: chartLabels.slice(-5),
            datasets: [{
              data: chartData.slice(-5),
              borderColor: '#a3e635',
              backgroundColor: 'rgba(163, 230, 53, 0.05)',
              tension: 0.3,
              fill: true,
              pointRadius: 3
            }]
          },
          options: {
            responsive: true,
            plugins: { legend: { display: false } },
            scales: {
              y: { grid: { color: '#27272a' }, ticks: { color: '#71717a', font: { size: 9 } } },
              x: { grid: { display: false }, ticks: { color: '#71717a', font: { size: 9 } } }
            }
          }
        });
      }
    }

    function getExerciseHistory(exerciseName) {
      return new Promise((resolve) => {
        const transaction = db.transaction(["exercises_history"], "readonly");
        const store = transaction.objectStore("exercises_history");
        const results = [];

        store.openCursor().onsuccess = (e) => {
          const cursor = e.target.result;
          if (cursor) {
            if (cursor.value.exerciseName === exerciseName) {
              results.push(cursor.value);
            }
            cursor.continue();
          } else {
            resolve(results);
          }
        };
      });
    }

    function getAllWorkouts() {
      return new Promise((resolve) => {
        const transaction = db.transaction(["workouts"], "readonly");
        const store = transaction.objectStore("workouts");
        const results = [];
        store.openCursor().onsuccess = (e) => {
          const cursor = e.target.result;
          if (cursor) {
            results.push(cursor.value);
            cursor.continue();
          } else {
            resolve(results);
          }
        };
      });
    }

    function copyToClipboard(escapedJson) {
      const workout = JSON.parse(decodeURIComponent(escapedJson));
      let text = `💪 NacorGym - ${workout.routine} (${workout.date})\n\n`;
      workout.exercises.forEach(ex => {
        text += `• ${ex.name}:\n`;
        ex.sets.forEach((set, sIdx) => {
          text += `  S${sIdx + 1}: ${set.peso} kg x ${set.reps} r\n`;
        });
        text += `\n`;
      });
      navigator.clipboard.writeText(text).then(() => alert("¡Copiado!"));
    }

    // --- COPIAS DE SEGURIDAD ---
    async function exportData() {
      const workouts = await getAllWorkouts();
      const transaction = db.transaction(["exercises_history"], "readonly");
      const store = transaction.objectStore("exercises_history");
      const history = [];

      store.openCursor().onsuccess = (e) => {
        const cursor = e.target.result;
        if (cursor) {
          history.push(cursor.value);
          cursor.continue();
        } else {
          const backup = { workouts, history };
          const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(backup));
          const dlAnchor = document.createElement('a');
          dlAnchor.setAttribute("href", dataStr);
          dlAnchor.setAttribute("download", `NacorGym_Backup.json`);
          dlAnchor.click();
        }
      };
    }

    function triggerImport() { document.getElementById('importFile').click(); }

    function importData(e) {
      const file = e.target.files[0];
      if (!file) return;

      const reader = new FileReader();
      reader.onload = async function(evt) {
        try {
          const backup = JSON.parse(evt.target.result);
          const transaction = db.transaction(["workouts", "exercises_history"], "readwrite");
          transaction.objectStore("workouts").clear();
          transaction.objectStore("exercises_history").clear();

          backup.workouts.forEach(w => transaction.objectStore("workouts").add(w));
          backup.history.forEach(h => transaction.objectStore("exercises_history").add(h));

          alert("¡Datos importados!");
          renderTabContent();
        } catch { alert("Error en el formato"); }
      };
      reader.readAsText(file);
    }
